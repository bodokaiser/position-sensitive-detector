\begin{align}
	I_+=I_\text{bias}+\frac{1}{2}I_\text{offset} &&
	I_\text{offset}=I_+-I_- \\
	I_-=I_\text{bias}-\frac{1}{2}I_\text{offset} &&
	I_\text{bias}=\frac{I_++I_-}{2}
\end{align}
